        // Función para actualizar estado de pedido
        async function updateOrderStatus(orderId, newStatus) {
            if (confirm(`¿Cambiar estado del pedido a "${newStatus}"?`)) {
                try {
                    const response = await fetch(`/admin/orders/${orderId}/status`, {
                        method: 'POST',
                        headers: {
                            'Content-Type': 'application/json',
                        },
                        body: JSON.stringify({ 
                            estado: newStatus,
                            notas_admin: `Estado cambiado a ${newStatus} desde dashboard` 
                        })
                    });
                    
                    if (response.ok) {
                        const result = await response.json();
                        if (result.success) {
                            alert('Estado actualizado correctamente');
                            location.reload();
                        } else {
                            alert('Error: ' + (result.message || 'No se pudo actualizar'));
                        }
                    } else {
                        alert('Error del servidor al actualizar estado');
                    }
                } catch (error) {
                    console.error('Error:', error);
                    alert('Error de conexión al actualizar estado');
                }
            }
        }
        
        // Función para enviar WhatsApp
        async function sendWhatsAppNotification(orderId) {
            const message = prompt('Escribe el mensaje para WhatsApp (opcional):', 
                'Hola, tu pedido ha sido actualizado. Gracias por tu compra.');
            
            if (message !== null) {
                try {
                    const response = await fetch(`/admin/orders/${orderId}/whatsapp`, {
                        method: 'POST',
                        headers: {
                            'Content-Type': 'application/json',
                        },
                        body: JSON.stringify({ mensaje: message })
                    });
                    
                    if (response.ok) {
                        const result = await response.json();
                        if (result.success) {
                            if (result.whatsapp_link) {
                                // Abrir enlace de WhatsApp
                                window.open(result.whatsapp_link, '_blank');
                            }
                            alert('Notificación enviada correctamente');
                        } else {
                            alert('Error: ' + (result.message || 'No se pudo enviar'));
                        }
                    } else {
                        alert('Error del servidor al enviar WhatsApp');
                    }
                } catch (error) {
                    console.error('Error:', error);
                    alert('Error de conexión al enviar WhatsApp');
                }
            }
        }
        
        // Auto-refresh cada 5 minutos para datos en tiempo real
        let autoRefresh = true;
        
        function toggleAutoRefresh() {
            autoRefresh = !autoRefresh;
            const btn = document.getElementById('refreshToggle');
            if (btn) {
                btn.innerHTML = autoRefresh ? 
                    '<i class="fas fa-pause"></i> Pausar Auto-Refresh' : 
                    '<i class="fas fa-play"></i> Activar Auto-Refresh';
                btn.className = autoRefresh ? 'btn btn-success btn-sm' : 'btn btn-secondary btn-sm';
            }
        }
        
        // Configurar auto-refresh si está activado
        if (autoRefresh) {
            setTimeout(() => {

                    location.reload();
                }
            }, 300000); // 5 minutos
        }
