    <script>
        // Funciones para el menú móvil
        function toggleSidebar() {
            const sidebar = document.querySelector('.admin-sidebar');
            sidebar.classList.toggle('active');
        }
        
        function closeSidebarOnMobile() {
            if (window.innerWidth <= 1024) {
                const sidebar = document.querySelector('.admin-sidebar');
                sidebar.classList.remove('active');
            }
        }
        
        // Cerrar sidebar al hacer clic en enlaces
        document.querySelectorAll('.admin-nav a, .logout-btn a').forEach(link => {
            link.addEventListener('click', closeSidebarOnMobile);
        });
        
        // Cerrar sidebar al hacer clic fuera (en móvil)
        document.addEventListener('click', (event) => {
            const sidebar = document.querySelector('.admin-sidebar');
            const mobileBtn = document.querySelector('.mobile-menu-btn');
            
            if (window.innerWidth <= 1024 && 
                sidebar.classList.contains('active') &&
                !sidebar.contains(event.target) &&
                !mobileBtn.contains(event.target)) {
                sidebar.classList.remove('active');
            }
        });
        
        // Funciones de filtrado
        function applyFilters() {
            const search = document.getElementById('search-input').value;
            const category = document.getElementById('category-filter').value;
            const status = document.getElementById('status-filter').value;
            
            let url = '/admin/products?';
            const params = [];
            
            if (search) params.push(`search=${encodeURIComponent(search)}`);
            if (category) params.push(`categoria=${encodeURIComponent(category)}`);
            if (status) params.push(`activo=${status}`);
            
            window.location.href = url + params.join('&');
        }
        
        function resetFilters() {
            window.location.href = '/admin/products';
        }
        
        function changePage(page) {
            const search = document.getElementById('search-input').value;
            const category = document.getElementById('category-filter').value;
            const status = document.getElementById('status-filter').value;
            
            let url = `/admin/products?page=${page}`;
            if (search) url += `&search=${encodeURIComponent(search)}`;
            if (category) url += `&categoria=${encodeURIComponent(category)}`;
            if (status) url += `&activo=${status}`;
            
            window.location.href = url;
        }
        
        // Funciones para productos
        function deleteProduct(id) {
            if (confirm('¿Estás seguro de desactivar este producto?')) {
                fetch(`/api/admin/products/${id}`, {
                    method: 'DELETE',
                    headers: {
                        'Content-Type': 'application/json'
                    }
                })
                .then(response => response.json())
                .then(data => {
                    if (data.success) {
                        alert('Producto desactivado exitosamente');
                        location.reload();
                    } else {
                        alert('Error: ' + data.message);
                    }
                })
                .catch(error => {
                    console.error('Error:', error);
                    alert('Error al desactivar el producto');
                });
            }
        }
        
        function restoreProduct(id) {
            if (confirm('¿Estás seguro de activar este producto?')) {
                fetch(`/api/admin/products/${id}/restore`, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json'
                    }
                })
                .then(response => response.json())
                .then(data => {
                    if (data.success) {
                        alert('Producto activado exitosamente');
                        location.reload();
                    } else {
                        alert('Error: ' + data.message);
                    }
                })
                .catch(error => {
                    console.error('Error:', error);
                    alert('Error al activar el producto');
                });
            }
        }
        
        function quickView(id) {
            fetch(`/api/admin/products/${id}/quick-view`)
                .then(response => response.json())
                .then(data => {
                    if (data.success) {
                        const product = data.product;
                        const modalContent = `
                            <div style="padding: 20px;">
                                <h3 style="margin-bottom: 15px;">${product.nombre}</h3>
                                <p><strong>ID:</strong> #${product.id}</p>
                                <p><strong>Categoría:</strong> ${product.categoria}</p>
                                <p><strong>Precio:</strong> €${parseFloat(product.precio).toFixed(2)}/${product.unidad}</p>
                                <p><strong>Stock:</strong> ${product.stock} unidades</p>
                                <p><strong>Estado:</strong> ${product.activo ? 'Activo' : 'Inactivo'}</p>
                                <p><strong>Destacado:</strong> ${product.destacado ? 'Sí' : 'No'}</p>
                                ${product.descripcion ? `<p><strong>Descripción:</strong> ${product.descripcion}</p>` : ''}
                            </div>
                        `;
                        
                        // Crear modal simple
                        const modal = document.createElement('div');
                        modal.style.position = 'fixed';
                        modal.style.top = '0';
                        modal.style.left = '0';
                        modal.style.width = '100%';
                        modal.style.height = '100%';
                        modal.style.backgroundColor = 'rgba(0,0,0,0.5)';
                        modal.style.display = 'flex';
                        modal.style.justifyContent = 'center';
                        modal.style.alignItems = 'center';
                        modal.style.zIndex = '1000';
                        
                        const modalBox = document.createElement('div');
                        modalBox.style.background = 'white';
                        modalBox.style.padding = '20px';
                        modalBox.style.borderRadius = '10px';
                        modalBox.style.maxWidth = '500px';
                        modalBox.style.width = '90%';
                        modalBox.style.position = 'relative';
                        
                        const closeBtn = document.createElement('button');
                        closeBtn.innerHTML = '×';
                        closeBtn.style.position = 'absolute';
                        closeBtn.style.top = '10px';
                        closeBtn.style.right = '10px';
                        closeBtn.style.background = 'none';
                        closeBtn.style.border = 'none';
                        closeBtn.style.fontSize = '24px';
                        closeBtn.style.cursor = 'pointer';
                        closeBtn.onclick = () => document.body.removeChild(modal);
                        
                        modalBox.innerHTML = modalContent;
                        modalBox.prepend(closeBtn);
                        modal.appendChild(modalBox);
                        
                        // Cerrar al hacer clic fuera
                        modal.onclick = (e) => {
                            if (e.target === modal) {
                                document.body.removeChild(modal);
                            }
                        };
                        
                        document.body.appendChild(modal);
                        
                    } else {
                        alert('Error: ' + data.message);
                    }
                })
                .catch(error => {
                    console.error('Error:', error);
                    alert('Error al cargar el producto');
                });
        }

        document.addEventListener('DOMContentLoaded', function() {
            // Buscar al presionar Enter
            document.getElementById('search-input').addEventListener('keypress', function(e) {
                if (e.key === 'Enter') {
                    applyFilters();
                }
            });
            
            // Ajustar tamaño del sidebar si es necesario
            const sidebar = document.querySelector('.admin-sidebar');
            const mainContent = document.querySelector('.admin-main');

                if (window.innerWidth > 1024) {
                    sidebar.style.transform = 'translateX(0)';
                    mainContent.style.marginLeft = '250px';
                    mainContent.style.width = 'calc(100% - 250px)';
                } else {
                    mainContent.style.marginLeft = '0';
                    mainContent.style.width = '100%';
                }
            }
            
            // Ajustar al cargar y al redimensionar
            adjustLayout();
            window.addEventListener('resize', adjustLayout);
        });
