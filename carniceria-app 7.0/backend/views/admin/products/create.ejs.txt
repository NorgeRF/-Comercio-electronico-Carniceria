    <script>
        // Funciones para el menú móvil
        function toggleSidebar() {
            const sidebar = document.querySelector('.admin-sidebar');
            sidebar.classList.toggle('active');
        }
        
        function closeSidebarOnMobile() {
            if (window.innerWidth <= 1024) {
                const sidebar = document.querySelector('.admin-sidebar');
                sidebar.classList.remove('active');
            }
        }
        
        // Cerrar sidebar al hacer clic en enlaces
        document.querySelectorAll('.admin-nav a, .logout-btn a').forEach(link => {
            link.addEventListener('click', closeSidebarOnMobile);
        });
        
        // Cerrar sidebar al hacer clic fuera (en móvil)
        document.addEventListener('click', (event) => {
            const sidebar = document.querySelector('.admin-sidebar');
            const mobileBtn = document.querySelector('.mobile-menu-btn');
            
            if (window.innerWidth <= 1024 && 
                sidebar.classList.contains('active') &&
                !sidebar.contains(event.target) &&
                !mobileBtn.contains(event.target)) {
                sidebar.classList.remove('active');
            }
        });
        
        // Validación del formulario
        document.getElementById('product-form').addEventListener('submit', function(e) {
            const nombre = document.getElementById('nombre').value;
            const precio = document.getElementById('precio').value;
            const categoria = document.getElementById('categoria').value;
            const unidad = document.getElementById('unidad').value;
            
            let isValid = true;
            let errorMessage = '';
            
            // Validar campos obligatorios
            if (!nombre.trim()) {
                isValid = false;
                errorMessage += '• El nombre es obligatorio\n';
            }
            
            if (!precio || parseFloat(precio) <= 0) {
                isValid = false;
                errorMessage += '• El precio debe ser mayor a 0\n';
            }
            
            if (!categoria) {
                isValid = false;
                errorMessage += '• La categoría es obligatoria\n';
            }
            
            if (!unidad) {
                isValid = false;
                errorMessage += '• La unidad es obligatoria\n';
            }
            
            if (!isValid) {
                e.preventDefault();
                alert('Por favor, corrige los siguientes errores:\n\n' + errorMessage);
                return false;
            }
            
            // Mostrar loading
            const submitBtn = this.querySelector('button[type="submit"]');
            const originalText = submitBtn.innerHTML;
            submitBtn.innerHTML = '<i class="fas fa-spinner fa-spin"></i> Procesando...';
            submitBtn.disabled = true;
        });
        
        // Función para eliminar producto
        function deleteProduct(id) {
            if (confirm('¿Estás seguro de eliminar este producto? Esta acción no se puede deshacer.')) {
                fetch(`/api/admin/products/${id}`, {
                    method: 'DELETE',
                    headers: {
                        'Content-Type': 'application/json'
                    }
                })
                .then(response => response.json())
                .then(data => {
                    if (data.success) {
                        alert('Producto eliminado exitosamente');
                        window.location.href = '/admin/products';
                    } else {
                        alert('Error: ' + data.message);
                    }
                })
                .catch(error => {
                    console.error('Error:', error);
                    alert('Error al eliminar el producto');
                });
            }
        }
        
        // Auto-guardado opcional (para formularios largos)
        let autoSaveTimer;
        
        function autoSave() {
            const form = document.getElementById('product-form');
            const formData = new FormData(form);
            const data = Object.fromEntries(formData);
            
            // Aquí podrías implementar auto-guardado local
            console.log('Auto-save:', data);
        }
        
        // Configurar auto-guardado cada 30 segundos
        document.querySelectorAll('input, textarea, select').forEach(element => {
            element.addEventListener('input', () => {
                clearTimeout(autoSaveTimer);
                autoSaveTimer = setTimeout(autoSave, 30000);
            });
        });
        
        // Prevenir pérdida de datos al salir
        window.addEventListener('beforeunload', function(e) {
            const form = document.getElementById('product-form');
            const hasChanges = form.checkValidity && form.checkValidity();
            
            if (hasChanges) {
                e.preventDefault();
                e.returnValue = 'Tienes cambios sin guardar. ¿Seguro que quieres salir?';
                return e.returnValue;
            }
        });
        
        // Ajustar layout al cargar y redimensionar
        document.addEventListener('DOMContentLoaded', function() {
            const sidebar = document.querySelector('.admin-sidebar');
            const mainContent = document.querySelector('.admin-main');
            
            function adjustLayout() {
                if (window.innerWidth > 1024) {
                    sidebar.style.transform = 'translateX(0)';
                    mainContent.style.marginLeft = '250px';
                    mainContent.style.width = 'calc(100% - 250px)';
                } else {
                    mainContent.style.marginLeft = '0';
                    mainContent.style.width = '100%';
                }
            }
            
            adjustLayout();
            window.addEventListener('resize', adjustLayout);
        });
